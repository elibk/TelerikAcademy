

function Solve(args) {

    var arr = arrToIntArr(args);
    var maxSum = getMaxSubArrSum(arr);
    return maxSum;
}


function getMaxSubArrSum(arr) {

    var sum = 0;
    var currentSum = 0;
    var i;
 
    for (i = 0; i < arr.length; i++) {
        currentSum += arr[i];
       

        if (currentSum < 0){
            currentSum = 0;
            
        }

        if (sum < currentSum){
            sum = currentSum;
        }

    }

    return sum;
}

function arrToIntArr(arr) {

    for (var i = 0; i < arr.length; i++) {
        arr[i] = parseInt(arr[i],10);
    }

    return arr;
}
